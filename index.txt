#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll n;
    cin >> n;
    map<ll, ll> freq;
    for (ll i = 1; i <= n; i += 1)
    {
        ll input;
        cin >> input;
        freq[i] = input;
    }
    ll ans = 0;
    for (ll i = 1; i <= n; i += 1)
    {
        ll temp = floor(freq[i] / 2.0);
        ans += temp;
        freq[i] = freq[i] - (2 * temp);
        if ((freq[i] > 0) && (freq[i + 1] > 0))
        {
            ans += 1;
            freq[i + 1] -= 1;
        }
    }
    cout << ans;
    return 0;
}
